\section{Pendahuluan}

% ==========================================
% SLIDE 2: LATAR BELAKANG MASALAH
% ==========================================
\begin{frame}[shrink=5]{Latar Belakang Masalah}{}
\begin{itemize}
  \item Transformasi digital telah mendorong adopsi enkripsi secara masif hingga melindungi sekitar 95\% dari total lalu lintas web saat ini.
  \item Peningkatan privasi ini sayangnya ikut dimanfaatkan oleh penjahat siber, di mana tercatat 85.9\% serangan siber kini disembunyikan di dalam saluran terenkripsi.
  \item Sistem keamanan jaringan konvensional seperti \textit{Deep Packet Inspection} (DPI) kehilangan efektivitasnya karena menjadi "buta" terhadap konten data yang terenkripsi.
  \item Solusi dekripsi di tengah jalan (\textit{Man-in-the-Middle}) juga tidak ideal karena memakan biaya komputasi yang sangat tinggi, merusak integritas data, dan memicu isu pelanggaran privasi.
\end{itemize}

\begin{alertblock}{Urgensi Penelitian}
Dibutuhkan metode klasifikasi dan pemantauan jaringan yang \textbf{\textit{payload-agnostic}} (mampu bekerja tanpa perlu mendekripsi data) untuk menjamin keamanan tanpa mengorbankan privasi pengguna.
\end{alertblock}
\end{frame}

% ==========================================
% SLIDE 3: KESENJANGAN, TUJUAN & BATASAN
% ==========================================
\begin{frame}{Kesenjangan, Tujuan, dan Batasan Penelitian}{}
\begin{itemize}
  \item Metode \textit{Encrypted Traffic Analysis} (ETA) saat ini masih kesulitan melakukan identifikasi \textit{fine-grained} pada level aplikasi karena mayoritas hanya mengandalkan satu jenis fitur ekstraksi.
  
  \item Mengatasi hal tersebut, penelitian ini bertujuan mengembangkan model representasi \textit{Hybrid Flow Vector} (HFV) untuk mencapai klasifikasi \textit{fine-grained} tanpa perlu mendekripsi \textit{payload} data.
  
  \item Ruang lingkup pengujian dievaluasi menggunakan dataset ISCX 2016 dengan kondisi \textit{flow} jaringan diasumsikan telah melalui tahap pembersihan (\textit{preprocessing}) yang ideal.
  
  \item Pemrosesan untuk ekstraksi fitur \textit{deep learning} dibatasi secara spesifik pada analisis 10 paket pertama dengan ukuran maksimum 784 \textit{byte} per paket.
\end{itemize}
\end{frame}

% SLIDE 4: USULAN SOLUSI (HFV)
% ==========================================
\begin{frame}[shrink=6]{Usulan Solusi: \textit{Hybrid Flow Vector} (HFV)}{}
\begin{columns}[T]
    
    % Kolom Kiri: Penjelasan Ringkas
    \begin{column}{0.48\textwidth}
    \vspace{0.5cm}
    \begin{itemize}
        \item HFV mengekstrak representasi jaringan menjadi \textbf{249 fitur} multi-modal secara \textit{payload-agnostic} (tanpa dekripsi) guna menjaga privasi pengguna.
        
        \item Pendekatan ini mengintegrasikan tiga pilar utama, yaitu fitur level \textit{byte} ($\alpha$), level \textit{flow} ($\beta$), dan level \textit{burst} ($\gamma$) ke dalam satu kesatuan $\Omega = \alpha \oplus \beta \oplus \gamma$.
        
        \item Ketiga pilar tersebut bekerja secara komplementer untuk menganalisis bentuk data mentah, statistik makro, dan ritme waktu pengiriman secara bersamaan tanpa perlu membuka isi pesan.
    \end{itemize}
    \end{column}

% SLIDE 13: KESIMPULAN
% ==========================================
\begin{frame}{Kesimpulan}{}

    % Baris 1: Teks 1 Poin Temuan Utama
    \begin{itemize}
        \item Kerangka kerja \textit{Hybrid Flow Vector} (HFV) terbukti sukses memberikan visibilitas jaringan secara \textit{payload-agnostic} dengan menggabungkan tiga pilar ortogonal untuk menciptakan \textit{signature} aplikasi yang sangat tangguh (\textit{evasion-resistant}).
    \end{itemize}
    
    \vspace{0.4cm}
    
    % Baris 2: Tiga Blok Penjelasan Ringkas
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.31\textwidth}
            \begin{block}{Akurasi Sangat Tinggi}
            Mencatatkan performa \textit{F1-Macro} 96.01\% (Biner), 91.84\% (Kategori), dan 91.62\% (Aplikasi) melalui arsitektur \textit{Ensemble Soft Voting}.
            \end{block}
        \end{column}
        \hfill
        \begin{column}{0.31\textwidth}
            \begin{block}{Sifat Komplementer}
            Statistik makroskopis sangat efektif untuk mendeteksi VPN, sementara \textit{Deep Learning} mutlak diperlukan untuk menembus klasifikasi \textit{fine-grained}.
            \end{block}
        \end{column}
        \hfill
        \begin{column}{0.31\textwidth}
            \begin{alertblock}{Ketahanan Arsitektur}
            Integrasi penuh ($\alpha \oplus \beta \oplus \gamma$) terbukti tidak sekadar redundan, melainkan menutupi kelemahan parameter statistik dari manipulasi penyerang.
            \end{alertblock}
        \end{column}
    \end{columns}

\end{frame}
